#End of Civil War - Colombia
country_event = {
	id = colombia.1
	title = colombia.1.t
	desc = colombia.1.d
	picture = GFX_report_event_colombia_war_end

	trigger = {
		original_tag = COL
		NOT = {
			country_exists = FAR
			country_exists = ELN
		}
		has_war = no
	}

	fire_only_once = yes
	
	option = {
		name = colombia.1.a

		add_political_power = 150
		add_stability = 0.1

		hidden_effect = {
			910 = {
				remove_core_of = FAR
			}
			1219 = {
				remove_core_of = ELN
			}
			news_event = {
				id = colombia_news.1
				days = 1
			}
		}

		ai_chance = {
			factor = 50
		}
	}
}
