		name = north_korea.9.a

			factor = 60

				factor = 2
				is_in_faction_with = FROM
			}
			
			#If North Korea is a democratic country, they have a high probability of support

				factor = 2
				FROM = { has_completed_focus = prk_lyuh_woon_hyung_s_unfulfilled_dream }
			}

			modifier = {
				factor = 0.5

					tag = KOR

				}
			}
			
			#If they military is weaker than North Korea's, they don't provide support.
			modifier = {
				factor = 0.8
				strength_ratio = { tag = FROM ratio > 1.5 }
			}
			modifier = {
				factor = 0.2
				strength_ratio = { tag = FROM ratio > 3 }
			}
			modifier = {
				factor = 0
				strength_ratio = { tag = FROM ratio > 4 }
			}
			
			#Anti-North Koreans
			modifier = {
				factor = 0.1
				OR = {
					has_country_leader = { ruling_only = yes name = "Lee Myung-bak" }
					has_country_leader = { ruling_only = yes name = "Park Geun-hye" }
				}
			}
			
		}
		FROM = { country_event = { id = north_korea.10 hours = 6 } }
		add_political_power = 50
		
		if = {
			limit = { ROOT = { tag = KOR } }
			custom_effect_tooltip = kor_increasing_influence_tt

				10 = { 
					PRK = { add_to_variable = { var = RegimeInfluence value = 4 } } 
					custom_effect_tooltip = kor_increasing_regimeInfluence_4_tt
				}
				40 = { 
					PRK = { add_to_variable = { var = RegimeInfluence value = 3 } } 
					custom_effect_tooltip = kor_increasing_regimeInfluence_3_tt
