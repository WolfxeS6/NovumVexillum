country_event = { # Take claimed states
	id = NVX_Warfaring.11
	hidden = yes
	
	is_triggered_only = yes
	
	option = {
		if = {
			limit = {
				FROM = {NOT = {has_war = yes}}
			}
			FROM = {
				random_owned_state = {
					limit = {
						any_neighbor_state = {
							is_owned_by = ROOT
						}
					}
					add_claim_by = ROOT
				}
			}
			create_wargoal = {
				type = take_state
				target = FROM
			}
		}	
	}
}
