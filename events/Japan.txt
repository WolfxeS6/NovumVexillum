#Invitation to the Japanese Alliance
country_event = {
	id = japan.1
	title = {
		text = japan.1.t1
		trigger = { FROM = { tag = JAP } }
	}
	title = {
		text = japan.1.t2
		trigger = {
			NOT = { FROM = { tag = JAP } }
		}
	}
	desc = {
		text = japan.1.d1
		trigger = { FROM = { tag = JAP } }
	}
	desc = {
		text = japan.1.d2
		trigger = { 
			NOT = { FROM = { tag = JAP } }
		}
	}
	picture = GFX_report_event_japanese_flag
	
	is_triggered_only = yes
	
	option = {	#We accept
		name = japan.1.o1
		set_country_flag = seek_japanese_alliance
		if = {
			limit = { JAP = { is_faction_leader = yes } }
			JAP = { add_to_faction = ROOT }
		}
		if = {
			limit = { JAP = { is_faction_leader = no } }
			random_country = {
				limit = {
					is_in_faction_with = JAP
					is_faction_leader = yes
				}
				add_to_faction = ROOT
			}
		}
		JAP = { country_event = diplomatic_response.1 }
		ai_chance = {
			factor = 10
			modifier = {
				has_similar_form_of_government = yes
				add = 50
			}
			modifier = {
				AND = {
					country_exists = PRC
					PRC = { has_government = communist }
					PRC = { is_puppet = no }
				}
				add = 50
			}
			modifier = {
				factor = 0
				is_neutral = yes
			}
		}
	}
	
	option = {	#We reject
		name = japan.1.o2
		JAP = { country_event = diplomatic_response.2 }
		ai_chance = {
			factor = 10
			modifier = {
				tag = KOR
				add = -10
			}
			modifier = {
				tag = CHI
				add = -10
			}
			modifier = {
				AND = {
					has_government = communist
					NOT = { tag = VIN }
				}
				add = 1000
			}
		}
	}
}
