	trigger = {
	    tag = URG
	    date > 1999.1.1
		date < 2004.12.31

		is_subject = no
		not = { has_country_flag = uru_election_2004 }
	}
	
	immediate = {
		set_country_flag = uru_election_2004
		promote_character = URG_jorge_larranaga
		promote_character = URG_guilhermo_stirling

		}

			popularity = 0.15
		}
