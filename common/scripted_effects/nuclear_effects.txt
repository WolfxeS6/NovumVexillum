NVX_nuclear_production_calc = {
    if = {
        limit = {has_country_flag = NV_nuclear_program}
        set_variable = {
            num_of_nuclear_reactor = 0
        }
        every_owned_state = {
            add_to_variable = {
                PREV.num_of_nuclear_reactor = building_level@nuclear_reactor
            }
        }
        set_variable = { NV_nuclear_production = num_of_nuclear_reactor}
    }
}
